904  size_t idx(fmap->second.get<0>());

910  std::sort(slinks.begin(),slinks.end(),[](link_t& m1,link_t& m2) {

913  }
914  );

919  deltaR.clear();

950  // calculate invariant mass
951  static bool invariantMass(const std::vector<fastjet::PseudoJet>& jets,double& minv,size_t& ijet,size_t& jjet)
952  {
953  typedef std::pair<size_t,size_t> link_t;
954  typedef std::map<double,link_t> linkmap_t;
955  // not enough jets
956  if ( jets.empty() ) { return false; }
957  if ( jets.size() == 1 ) { ijet = 0; jjet = ijet; minv = 0.; return false; }
958  if ( jets.size() == 2 ) { ijet = 0; jjet = 1; minv = (jets.front()+jets.back()).m(); return true; }
959  // find the highest invariant mass between any N>2 jets
960  linkmap_t orderedLinks;
961  if ( Analysis::Select::Event::minMinv > Analysis::Select::Event::maxMinv ) {
962  for ( size_t i(0); i<jets.size()-1; ++i ) {
963  for ( size_t j(i+1); j<jets.size(); ++j ) {
964  double m((jets.at(i)+jets.at(j)).m());
965  if ( m > Analysis::Select::Event::minMinv ) { orderedLinks[m] = link_t(i,j); }
966  }
967  }
968  } else {
969  for ( size_t i(0); i<jets.size()-1; ++i ) {
970  for ( size_t j(i+1); j<jets.size(); ++j ) {
971  double m((jets.at(i)+jets.at(j)).m());
972  if ( m > Analysis::Select::Event::minMinv && m < Analysis::Select::Event::maxMinv ) { orderedLinks[m] = link_t(i,j); }
973  }
974  }
975  }
976  // return the findings
977  if ( orderedLinks.empty() ) { return false; }
978  auto fmap = orderedLinks.rbegin();
979  ijet = fmap->second.first;
980  jjet = fmap->second.second;
981  minv = fmap->first;
982  return true;
983  }
